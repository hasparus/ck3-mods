namespace = vigil_holy_site

# Vigil at the Holy Site — first step: choose the officiant of the rite.
vigil_holy_site.0002 = {
	type = character_event
	title = vigil_holy_site.0002.t
	desc = vigil_holy_site.0002.desc
	theme = faith
	override_background = {
		trigger = { faith = { religion_tag = christianity_religion } }
		reference = temple_church
	}
	override_background = {
		trigger = { faith = { religion_tag = islam_religion } }
		reference = temple_mosque
	}
	override_background = {
		trigger = { faith = { religion_tag = tengri_religion } }
		reference = mpo_temple_steppe
	}
	override_background = { reference = holy_site_generic }

	left_portrait = {
		character = root
		animation = personality_zealous
	}
	right_portrait = {
		trigger = { exists = scope:vigil_chaplain }
		character = scope:vigil_chaplain
		animation = personality_zealous
	}
	lower_right_portrait = {
		trigger = { exists = scope:vigil_captive_priest }
		character = scope:vigil_captive_priest
		animation = shame
	}

	immediate = {
		if = {
			limit = { exists = cp:councillor_court_chaplain }
			cp:councillor_court_chaplain = { save_scope_as = vigil_chaplain }
		}
		ordered_prisoner = {
			limit = {
				NOT = { faith = root.faith }
				OR = {
					government_has_flag = government_is_theocracy
					has_council_position = councillor_court_chaplain
					has_trait = theologian
					has_trait = devoted
					has_trait = saint
					has_trait = witch
				}
			}
			order_by = learning
			save_scope_as = vigil_captive_priest
		}
	}

	# Yourself
	option = {
		name = vigil_holy_site.0002.a
		save_scope_as = vigil_officiant
		add_stress = minor_stress_gain
		trigger_event = vigil_holy_site.0003
	}

	# Court chaplain
	option = {
		name = vigil_holy_site.0002.b
		trigger = {
			exists = scope:vigil_chaplain
			gold >= 50
		}
		scope:vigil_chaplain = { save_scope_as = vigil_officiant }
		add_gold = -50
		scope:vigil_chaplain = { add_gold = 50 }
		stress_impact = { greedy = minor_stress_gain }
		trigger_event = vigil_holy_site.0003
	}

	# Captive priest of a different faith
	option = {
		name = vigil_holy_site.0002.c
		trigger = { exists = scope:vigil_captive_priest }
		scope:vigil_captive_priest = { save_scope_as = vigil_officiant }
		if = {
			limit = {
				OR = {
					has_trait = sadistic
					has_trait = arbitrary
				}
			}
			add_stress = minor_stress_loss
		}
		else = {
			random_list = {
				15 = { add_trait = sadistic }
				15 = { add_trait = arbitrary }
				70 = {}
			}
		}
		trigger_event = vigil_holy_site.0003
	}

	# Step away from the rite entirely.
	option = {
		name = vigil_holy_site.0002.d
	}
}

# Vigil at the Holy Site — third step: read the omens of the rite.
# Outcome is a Learning-weighted random_list on the officiant's stat.
vigil_holy_site.0004 = {
	type = character_event
	title = vigil_holy_site.0004.t
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = { has_variable_value = { name = vigil_omens_result value = flag:crit } }
				desc = vigil_holy_site.0004.desc.crit
			}
			triggered_desc = {
				trigger = { has_variable_value = { name = vigil_omens_result value = flag:success } }
				desc = vigil_holy_site.0004.desc.success
			}
			triggered_desc = {
				trigger = { has_variable_value = { name = vigil_omens_result value = flag:ambiguous } }
				desc = vigil_holy_site.0004.desc.ambiguous
			}
			desc = vigil_holy_site.0004.desc.failure
		}
	}
	theme = faith
	override_background = {
		trigger = { faith = { religion_tag = christianity_religion } }
		reference = temple_church
	}
	override_background = {
		trigger = { faith = { religion_tag = islam_religion } }
		reference = temple_mosque
	}
	override_background = {
		trigger = { faith = { religion_tag = tengri_religion } }
		reference = mpo_temple_steppe
	}
	override_background = { reference = holy_site_generic }

	left_portrait = {
		character = root
		animation = personality_zealous
	}
	right_portrait = {
		trigger = {
			exists = scope:vigil_officiant
			NOT = { scope:vigil_officiant = root }
		}
		character = scope:vigil_officiant
		animation = personality_zealous
	}

	immediate = {
		random_list = {
			15 = {
				modifier = {
					trigger = { scope:vigil_officiant = { learning >= 14 } }
					factor = {
						value = scope:vigil_officiant.learning
						subtract = 13
						multiply = 4
					}
				}
				set_variable = { name = vigil_omens_result value = flag:crit }
				add_character_modifier = {
					modifier = vigil_blessed_by_omens
					years = 5
				}
				faith = { add_fervor = 3 }
				random_targeting_scheme = {
					limit = { scheme_type = murder }
					expose_scheme = yes
				}
			}
			30 = {
				modifier = {
					trigger = { scope:vigil_officiant = { learning >= 10 } }
					add = {
						value = scope:vigil_officiant.learning
						subtract = 10
						multiply = 3
					}
				}
				modifier = {
					trigger = { scope:vigil_officiant = { learning <= 5 } }
					add = 5
				}
				set_variable = { name = vigil_omens_result value = flag:success }
				add_character_modifier = {
					modifier = vigil_blessed_by_omens
					years = 5
				}
			}
			35 = {
				modifier = {
					trigger = { scope:vigil_officiant = { learning >= 12 } }
					factor = 0.5
				}
				modifier = {
					trigger = { scope:vigil_officiant = { learning <= 6 } }
					add = 10
				}
				set_variable = { name = vigil_omens_result value = flag:ambiguous }
			}
			50 = {
				modifier = {
					trigger = { scope:vigil_officiant = { learning >= 15 } }
					factor = 0
				}
				modifier = {
					trigger = { scope:vigil_officiant = { learning <= 4 } }
					add = 20
				}
				set_variable = { name = vigil_omens_result value = flag:failure }
				add_character_modifier = {
					modifier = vigil_dark_omens
					years = 3
				}
				add_stress = minor_stress_gain
				scope:vigil_officiant = {
					random = {
						chance = 25
						modifier = { factor = 0 has_trait = paranoid }
						add_trait = paranoid
					}
				}
			}
		}
	}

	option = {
		name = vigil_holy_site.0004.a
		remove_variable = vigil_omens_result
	}
}

# Burn the Infidel Holy Place — confirmation window for the desecration decision.
vigil_holy_site.0001 = {
	type = character_event
	title = vigil_holy_site.0001.t
	desc = vigil_holy_site.0001.desc
	theme = faith
	override_background = { reference = burning_building }

	left_portrait = {
		character = root
		animation = personality_zealous
	}

	immediate = {
		random_army = {
			limit = {
				location.barony = {
					is_holy_site = yes
					NOT = { is_holy_site_of = root.faith }
					OR = {
						barony_controller ?= {
							OR = {
								this = root
								target_is_liege_or_above = root
							}
						}
						var:vigil_last_raider ?= root
					}
				}
			}
			location.barony = { save_scope_as = vigil_desecrated_barony }
			location.county = { save_scope_as = vigil_desecrated_county }
			if = {
				limit = {
					location.barony.holder ?= {
						OR = {
							this = root
							target_is_liege_or_above = root
						}
					}
				}
				location.county.de_jure_liege = { save_scope_as = vigil_reclaim_duchy }
			}
		}
		scope:vigil_desecrated_county.faith = { save_scope_as = vigil_harmed_faith }
	}

	option = {
		name = vigil_holy_site.0001.a

		scope:vigil_desecrated_county = {
			set_variable = { name = desecrator value = root }
			add_county_modifier = {
				modifier = zz_holy_site_desecrated
				years = 80
			}
		}
		add_piety = 5000

		# Plotter: prefer the highest-intrigue existing rival of root in the harmed-faith pool;
		# fall back to the highest-intrigue character otherwise.
		ordered_vassal_or_below = {
			limit = {
				faith = scope:vigil_harmed_faith
				has_relation_rival = root
			}
			order_by = intrigue
			save_scope_as = vigil_plotter
		}
		ordered_neighboring_top_liege_realm_owner = {
			limit = {
				faith = scope:vigil_harmed_faith
				has_relation_rival = root
				OR = {
					NOT = { exists = scope:vigil_plotter }
					intrigue > scope:vigil_plotter.intrigue
				}
			}
			order_by = intrigue
			save_scope_as = vigil_plotter
		}
		if = {
			limit = { NOT = { exists = scope:vigil_plotter } }
			ordered_vassal_or_below = {
				limit = { faith = scope:vigil_harmed_faith }
				order_by = intrigue
				save_scope_as = vigil_plotter
			}
			ordered_neighboring_top_liege_realm_owner = {
				limit = {
					faith = scope:vigil_harmed_faith
					OR = {
						NOT = { exists = scope:vigil_plotter }
						intrigue > scope:vigil_plotter.intrigue
					}
				}
				order_by = intrigue
				save_scope_as = vigil_plotter
			}
		}

		# Avenger: prefer the strongest existing rival of root in the harmed-faith pool;
		# fall back to the strongest landed character otherwise.
		ordered_vassal_or_below = {
			limit = {
				faith = scope:vigil_harmed_faith
				is_landed = yes
				has_relation_rival = root
			}
			order_by = max_military_strength
			save_scope_as = vigil_avenger
		}
		ordered_neighboring_top_liege_realm_owner = {
			limit = {
				faith = scope:vigil_harmed_faith
				has_relation_rival = root
				OR = {
					NOT = { exists = scope:vigil_avenger }
					max_military_strength > scope:vigil_avenger.max_military_strength
				}
			}
			order_by = max_military_strength
			save_scope_as = vigil_avenger
		}
		if = {
			limit = { NOT = { exists = scope:vigil_avenger } }
			ordered_vassal_or_below = {
				limit = {
					faith = scope:vigil_harmed_faith
					is_landed = yes
				}
				order_by = max_military_strength
				save_scope_as = vigil_avenger
			}
			ordered_neighboring_top_liege_realm_owner = {
				limit = {
					faith = scope:vigil_harmed_faith
					OR = {
						NOT = { exists = scope:vigil_avenger }
						max_military_strength > scope:vigil_avenger.max_military_strength
					}
				}
				order_by = max_military_strength
				save_scope_as = vigil_avenger
			}
		}

		if = {
			limit = { exists = scope:vigil_plotter }
			scope:vigil_plotter = {
				set_relation_rival = {
					reason = rival_zz_holy_site_desecrated
					target = root
				}
				start_scheme = {
					type = murder
					target = root
				}
			}
		}

		if = {
			limit = { exists = scope:vigil_avenger }
			scope:vigil_avenger = {
				set_relation_rival = {
					reason = rival_zz_holy_site_desecrated
					target = root
				}
				if = {
					limit = { exists = scope:vigil_reclaim_duchy }
					set_variable = {
						name = vigil_reclaim_duchy
						value = scope:vigil_reclaim_duchy
					}
				}
				start_war = {
					casus_belli = zz_holy_site_desecrated_punitive_war_cb
					target = root
				}
			}
		}
	}

	option = {
		name = vigil_holy_site.0001.b
	}
}
