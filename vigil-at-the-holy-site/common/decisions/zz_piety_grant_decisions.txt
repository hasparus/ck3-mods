zz_holy_site_vigil_decision = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/zz_holy_site_vigil_decision.dds"
	}

	decision_group_type = minor

	ai_check_interval = 0

	is_shown = {
		is_ai = no
		NOT = { has_global_variable = zz_holy_site_vigil_done }
	}

	is_valid = {
		custom_tooltip = {
			text = zz_holy_site_vigil_requires_holy_site_tt
			any_sub_realm_barony = {
				is_holy_site_of = root.faith
			}
		}
		custom_tooltip = {
			text = zz_holy_site_vigil_requires_prisoner_tt
			any_prisoner = { count >= 1 }
		}
	}

	effect = {
		trigger_event = vigil_holy_site.0002
	}

	ai_will_do = {
		value = 0
	}
}

zz_holy_site_desecration_decision = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/zz_holy_site_desecration_decision.dds"
	}

	decision_group_type = minor

	ai_check_interval = 0

	is_shown = {
		is_ai = no
		has_global_variable = zz_holy_site_vigil_done
	}

	is_valid = {
		custom_tooltip = {
			text = zz_holy_site_desecration_requires_army_tt
			any_army = {
				location.barony = {
					is_holy_site = yes
					NOT = { is_holy_site_of = root.faith }
					OR = {
						barony_controller ?= {
							OR = {
								this = root
								target_is_liege_or_above = root
							}
						}
						var:vigil_last_raider ?= root
					}
				}
			}
		}
	}

	effect = {
		trigger_event = vigil_holy_site.0001
	}

	ai_will_do = {
		value = 0
	}
}
