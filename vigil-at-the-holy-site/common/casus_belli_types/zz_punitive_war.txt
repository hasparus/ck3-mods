zz_punitive_war_cb = {
	icon = invasion
	group = invasion

	target_titles = none
	target_de_jure_regions_above = no

	allow_hostages = no
	allowed_for_character = {}
	allowed_against_character = {}

	cost = {}

	attacker_wargoal_percentage = 0.8

	on_declaration = {
		on_declared_war = yes
	}

	should_invalidate = {
		scope:defender = { is_alive = no }
	}

	on_victory_desc = {
		desc = zz_punitive_war_victory_desc
	}

	on_victory = {
		scope:attacker = {
			show_pow_release_message_effect = yes
			add_piety = 1500
			add_prestige = 500
		}
		scope:defender = {
			imprison = {
				target = scope:attacker
				type = dungeon
			}
			add_piety = -500
			add_prestige = -1000
		}
		if = {
			limit = { exists = scope:attacker.var:vigil_reclaim_duchy }
			create_title_and_vassal_change = {
				type = conquest
				save_scope_as = change
			}
			scope:attacker.var:vigil_reclaim_duchy = {
				change_title_holder = {
					holder = scope:attacker
					change = scope:change
					take_baronies = yes
				}
			}
			resolve_title_and_vassal_change = scope:change
			scope:attacker = { remove_variable = vigil_reclaim_duchy }
		}
	}

	on_white_peace = {
		scope:attacker = { show_pow_release_message_effect = yes }
	}

	on_defeat = {
		scope:attacker = {
			show_pow_release_message_effect = yes
			add_piety = -500
		}
		scope:defender = {
			add_piety = 200
		}
	}

	on_primary_attacker_death = inherit
	on_primary_defender_death = inherit
	transfer_behavior = transfer

	war_name = "ZZ_PUNITIVE_WAR_NAME"

	interface_priority = 80

	max_attacker_score_from_battles = 100
	max_defender_score_from_battles = 100
	max_attacker_score_from_occupation = 150
	max_defender_score_from_occupation = 150
}
