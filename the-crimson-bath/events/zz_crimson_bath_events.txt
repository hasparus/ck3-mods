namespace = crimson_bath

# The Crimson Bath — one screen per captive, one captive per malady.
# Recurses on itself until either the maladies or the eligible captives
# run out, then triggers the aftermath via a shared scripted effect.
crimson_bath.0001 = {
	type = character_event
	title = crimson_bath_decision
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = { has_variable = crimson_bath_in_progress }
				desc = crimson_bath.0001.desc.position.next
			}
			desc = crimson_bath.0001.desc.position.first
		}
		first_valid = {
			triggered_desc = {
				trigger = { has_trait = infirm }
				desc = crimson_bath.0001.desc.malady.infirm
			}
			triggered_desc = {
				trigger = { has_trait = clouded_eyes }
				desc = crimson_bath.0001.desc.malady.clouded_eyes
			}
			triggered_desc = {
				trigger = { has_trait = faltering_heart }
				desc = crimson_bath.0001.desc.malady.faltering_heart
			}
			triggered_desc = {
				trigger = { has_trait = fragile_bones }
				desc = crimson_bath.0001.desc.malady.fragile_bones
			}
			desc = crimson_bath.0001.desc.malady.withering_mind
		}
		triggered_desc = {
			trigger = { has_trait = sadistic }
			desc = crimson_bath.0001.desc.sadistic
		}
	}
	theme = dread
	override_background = { reference = dungeon }

	left_portrait = {
		character = root
		animation = personality_callous
	}
	right_portrait = {
		trigger = { exists = scope:crimson_bath_offering }
		character = scope:crimson_bath_offering
		animation = shame
	}

	immediate = {
		random_prisoner = {
			limit = { age < 30 }
			save_scope_as = crimson_bath_offering
		}
	}

	# Open the vein on this captive: kill, kin grief, remove one malady,
	# stress relief if callous or sadistic, then recurse or finish.
	option = {
		name = crimson_bath.0001.a

		scope:crimson_bath_offering = {
			every_close_family_member = {
				add_opinion = {
					modifier = crimson_bath_kin_grief
					target = root
					years = 15
				}
			}
			death = {
				killer = root
				reason = death_sacrificed
			}
		}

		if = {
			limit = {
				OR = {
					has_trait = callous
					has_trait = sadistic
				}
			}
			add_stress = minor_stress_loss
		}

		set_variable = { name = crimson_bath_in_progress }

		if = {
			limit = { has_trait = infirm }
			remove_trait = infirm
		}
		else_if = {
			limit = { has_trait = clouded_eyes }
			remove_trait = clouded_eyes
		}
		else_if = {
			limit = { has_trait = faltering_heart }
			remove_trait = faltering_heart
		}
		else_if = {
			limit = { has_trait = fragile_bones }
			remove_trait = fragile_bones
		}
		else_if = {
			limit = { has_trait = withering_mind }
			remove_trait = withering_mind
		}

		# Track maladies undone across all baths. On the tenth, schedule the
		# transformation event ten years out.
		change_variable = { name = crimson_bath_heals_count add = 1 }
		if = {
			limit = { var:crimson_bath_heals_count = 10 }
			trigger_event = { id = crimson_bath.0003 years = 10 }
		}

		# 10% chance per sacrifice to interrupt with the Numb screen, but only
		# for faiths that don't already bless ritual sacrifice, and only while
		# root carries neither callous nor sadistic. The Numb screen lands
		# inline (it adds callous + stress and then continues the chain), so
		# subsequent kills naturally fail the gate.
		if = {
			limit = {
				NOR = {
					has_trait = callous
					has_trait = sadistic
				}
				faith = { NOT = { has_doctrine = tenet_human_sacrifice } }
			}
			random_list = {
				10 = { trigger_event = crimson_bath.0002 }
				90 = { crimson_bath_continue_or_finish_effect = yes }
			}
		}
		else = {
			crimson_bath_continue_or_finish_effect = yes
		}
	}

	# Stop the rite here: no more sacrifices, but the maladies already
	# undone stay undone and the aftermath still lands.
	option = {
		name = crimson_bath.0001.b
		crimson_bath_aftermath_effect = yes
	}
}

# The Numb — short follow-up that fires with 10% chance after the
# rite if root is not already callous. Grants the trait and meaningful
# stress relief.
crimson_bath.0002 = {
	type = character_event
	title = crimson_bath.0002.t
	desc = crimson_bath.0002.desc
	theme = dread
	override_background = { reference = bedchamber }

	left_portrait = {
		character = root
		animation = personality_callous
	}

	option = {
		name = crimson_bath.0002.a
		add_trait = callous
		add_stress = medium_stress_loss
		crimson_bath_continue_or_finish_effect = yes
	}
}

# Ten captives bled, ten years on. The body has been remade, and the long
# wait has been paid for in the night. Each option is the player learning
# what they have been doing without choosing it; the kill or the modifier
# is where the count fell.
crimson_bath.0003 = {
	type = character_event
	title = crimson_bath.0003.t
	desc = crimson_bath.0003.desc
	theme = dread
	override_background = { reference = bedchamber }

	left_portrait = {
		character = root
		animation = shock
	}
	right_portrait = {
		trigger = { exists = scope:crimson_bath_corridor_victim }
		character = scope:crimson_bath_corridor_victim
		animation = personality_compassionate
	}

	immediate = {
		random_held_title = {
			limit = {
				tier = tier_county
				NOT = { this = root.capital_county }
			}
			save_scope_as = crimson_bath_village
		}
		random_courtier = {
			limit = {
				is_adult = yes
				is_landed = no
			}
			save_scope_as = crimson_bath_corridor_victim
		}
	}

	# Did I prowl in [village]?
	option = {
		name = crimson_bath.0003.a
		trigger = {
			exists = scope:crimson_bath_village
		}
		scope:crimson_bath_village = {
			add_county_modifier = {
				modifier = crimson_bath_hunting_grounds
				years = 10
			}
		}
		add_trait = strong
		add_trait = albino
	}

	# Have I visited [name] last night?
	option = {
		name = crimson_bath.0003.b
		trigger = {
			exists = scope:crimson_bath_corridor_victim
		}
		scope:crimson_bath_corridor_victim = {
			death = {
				killer = root
				reason = death_murder_known
			}
		}
		add_trait = strong
		add_trait = albino
	}

}
