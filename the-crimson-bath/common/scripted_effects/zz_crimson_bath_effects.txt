# Shared aftermath for the Crimson Bath: witch branding, the closing
# stress, and the nickname roll.
crimson_bath_aftermath_effect = {
	if = {
		limit = { NOT = { has_trait = witch } }
		give_witch_secret_or_trait_effect = yes
	}
	add_stress = minor_stress_gain
	if = {
		limit = { has_any_nickname = no }
		random_list = {
			50 = { give_nickname = nick_the_vampire }
			15 = { give_nickname = nick_the_red }
			15 = { give_nickname = nick_the_strigoi }
			10 = { give_nickname = nick_the_undying }
			10 = { give_nickname = nick_the_pale }
		}
	}
	remove_variable = crimson_bath_in_progress
}
